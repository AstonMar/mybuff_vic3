						has_diplomatic_pact = { who = root type = tributary }		#是你的朝贡国
					}
				}
			}
			#变成傀儡国
			create_diplomatic_pact = {
				country = scope:target_country
				type = vassal
			}
		}

					OR = {

						has_diplomatic_pact = { who = root type = tributary }		#是你的朝贡国
					}
				}
			}

				country = scope:target_country
				type = puppet
			}
		}

		}

	type = investment

	icon = "gfx/interface/icons/company_charter_icons/investment_charter.dds"

		company_prosperity >= 50
		owner = {

				any_scope_treaty = {

				}
				num_subjects >= 1
			}
		}
	}

	ai_weight = {
        value = 0//删掉Ai投资权倾向防止区域公司生成
    }

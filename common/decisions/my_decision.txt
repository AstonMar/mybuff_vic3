add_my_buff_decision = {
	is_shown = {
		NOT = {
			has_modifier = my_buff
		}
		is_player = yes
	}
	when_taken = {
		add_modifier = {
			name = my_buff
		}
	}
	ai_chance = {
		base = 0
	}
}

remove_my_buff_decision = {
	is_shown = {
		has_modifier = my_buff
		is_player = yes
	}
	when_taken = {
		remove_modifier = my_buff
	}
	ai_chance = {
		base = 0
	}
}
